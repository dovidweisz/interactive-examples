const object = { 'foo': 'a', 'bar': 2};
const entries = Object.entries(object);

console.log(entries);
// expected output: Array [Array ["foo", "a"], Array ["bar", 2]]
// order is not guaranteed

const sorted = [...entries].sort();
console.log(sorted);
// expected output: Array [Array ["bar", 2], Array ["foo", "a"]]

for (let [key, value] of sorted) {  
    console.log(`${key}: ${value}`);
}
// expected output:
// "bar: 2"
// "foo: a"
